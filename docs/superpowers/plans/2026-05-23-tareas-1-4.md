```typescript
/**
 * Envía email al planner notificando que hay una tarea BORRADOR lista para planificar.
 * plannerEmail: email del usuario planner (obtenido del frontend vía selector o hardcoded
 * temporalmente desde companies.planner_email o profiles con rol supervisor).
 */
export async function notificarPlanner(cotizacion_id: string, plannerEmail: string): Promise<ActionResponse> {
    const { adminClient, tenantId } = await getSupabaseContext()
    if (!adminClient || !tenantId) return { message: 'No autorizado' }

    const tareaInfo = await getCotizacionTareaInfo(cotizacion_id)
    if (!tareaInfo) return { message: 'No hay tarea generada para esta cotización' }

    // Importar el helper de email del proyecto
    const { sendEmail } = await import('@/lib/email')

    const webUrl = process.env.NEXT_PUBLIC_APP_URL ?? 'https://reporta.app'
    const planLink = `${webUrl}/planificacion?tarea=${tareaInfo.id}`

    const result = await sendEmail({
        to: plannerEmail,
        subject: `[Reporta] Nueva tarea para planificar: ${tareaInfo.codigo ?? tareaInfo.id.slice(0, 8)}`,
        html: `
            <p>Hola,</p>
            <p>La cotización generó una tarea en estado <strong>BORRADOR</strong> que requiere fechas y recursos:</p>
            <ul>
                <li><strong>Tarea:</strong> ${tareaInfo.codigo} — ${tareaInfo.titulo}</li>
            </ul>
            <p><a href="${planLink}">Ir a Planificación →</a></p>
            <p style="color:#6b7280;font-size:12px">Este correo fue enviado automáticamente por Reporta.</p>
        `,
    })

    if (!result.success) return { message: result.message ?? 'Error al enviar email' }
    return { success: true, message: 'Planner notificado por correo' }
}
```

```typescript
// lib/queries/reportes-lista.ts
import { supabase } from '@/lib/supabase'
import { getCurrentTenantId } from '@/lib/tenant'
import { useAuth } from '@/lib/auth'

export type ReporteMaquinariaResumen = {
    id: string
    codigo_documento: string | null
    fecha: string
    estado_sync: string | null
    pdf_url: string | null
    horas_trabajadas: number | null
    maquinaria: { nombre: string; codigo_interno: string | null } | null
    tarea: { codigo: string; titulo: string } | null
}

export type ReportePersonalResumen = {
    id: string
    fecha: string
    horas_trabajadas: number | null
    pdf_url: string | null
    trabajo_realizado: string | null
    tarea: { codigo: string; titulo: string } | null
}

export async function getReportesMaquinariaDelMes(
    userId: string,
    mes: number,
    anio: number,
): Promise<ReporteMaquinariaResumen[]> {
    const tenantId = await getCurrentTenantId()
    if (!tenantId) return []

    const desde = `${anio}-${String(mes).padStart(2, '0')}-01`
    const hasta = new Date(anio, mes, 0).toISOString().split('T')[0] // último día del mes

    const { data, error } = await supabase
        .from('reportes_maquinaria')
        .select(`
            id, codigo_documento, fecha, estado_sync, pdf_url, horas_trabajadas,
            maquinaria:maquinarias(nombre, codigo_interno),
            tarea:tareas(codigo, titulo)
        `)
        .eq('tenant_id', tenantId)
        .eq('operador_id', userId)
        .gte('fecha', desde)
        .lte('fecha', hasta)
        .order('fecha', { ascending: false })

    if (error) { console.warn('[reportes-lista]', error.message); return [] }
    return (data ?? []) as ReporteMaquinariaResumen[]
}

export async function getReportesPersonalDelMes(
    userId: string,
    mes: number,
    anio: number,
): Promise<ReportePersonalResumen[]> {
    const tenantId = await getCurrentTenantId()
    if (!tenantId) return []

    const desde = `${anio}-${String(mes).padStart(2, '0')}-01`
    const hasta = new Date(anio, mes, 0).toISOString().split('T')[0]

    const { data, error } = await supabase
        .from('reportes_personal')
        .select(`
            id, fecha, horas_trabajadas, pdf_url, trabajo_realizado,
            tarea:tareas(codigo, titulo)
        `)
        .eq('tenant_id', tenantId)
        .eq('created_by', userId)
        .gte('fecha', desde)
        .lte('fecha', hasta)
        .order('fecha', { ascending: false })

    if (error) { console.warn('[reportes-lista]', error.message); return [] }
    return (data ?? []) as ReportePersonalResumen[]
}
```

```typescript
import { NextRequest, NextResponse } from 'next/server'
import { createAdminClient } from '@/utils/supabase/admin'

export async function POST(
    req: NextRequest,
    { params }: { params: { id: string } }
) {
    const serviceKey = req.headers.get('X-Report-Service-Key')
    if (serviceKey !== process.env.REPORT_SERVICE_KEY) {
        return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })
    }

        .eq('id', params.id)

    return NextResponse.json({ ok: true, message: 'PDF queued' })
}
```
